[ -d "$HOME" ]; then echo '> sudo mkdir' "$HOME" sudo mkdir "$HOME" fi fi GROUP="" if ! chown $USER$GROUP "$HOME" &> /dev/null; then echo '>' sudo chown $USER$GROUP "$HOME" sudo chown $USER$GROUP "$HOME" fi mkdir -p "$HOME/build/abepralle/morlock" if ! "/opt/morlock/build/abepralle/morlock/morlock" bootstrap
